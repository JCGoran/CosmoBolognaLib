56 void cbl::data::Data1D::read (const string input_file, const int skip_nlines)
57 {
58  ifstream fin(input_file.c_str()); checkIO(fin, input_file);
59  string line;
60 
61  bool moreColumns = true;
62  unsigned int col = 3, ind1 = 1, ind2 = 2;
63 
64  // enter here the first time, and continue to read in case of more the 3 columns (e.g. for clustering multipoles)
65  while (moreColumns) {
66 
67  // skip the first nlines (in case of header lines)
68  if (skip_nlines>0)
69  for (int i=0; i<skip_nlines; ++i)
70  getline(fin, line);
71 
72  // read the file lines
73  while (getline(fin, line)) {
74 
75  stringstream ss(line);
76  vector<double> num; double NUM;
77  while (ss>>NUM) num.emplace_back(NUM);
78 
79  if ((int)num.size()<2) ErrorCBL("Error in cbl::data::Data1D::read(): the input file has less than 2 columns!");
80 
81  // if there are more than 3 columns then the next columns will
82  // be read and added to the first one
83  if (num.size()<=col) moreColumns = false;
84 
85  m_x.emplace_back(num[0]);
86  m_data.emplace_back(num[ind1]);
87  m_error.emplace_back((num.size()>ind2) ? num[ind2] : 1);
88 
89  }
90 
91  // go back at the beginning of the file
92  fin.clear(); fin.seekg(ios::beg);
93 
94  col += 2;
95  ind1 += 2;
96  ind2 += 2;
97  }
98 
99  fin.clear(); fin.close();
100 
101 
102  // set the data type and diagonal covariance
103 
104  m_ndata = m_data.size();
105 
106  m_covariance.resize(m_ndata, vector<double>(m_ndata, 0));
107  for (int i=0; i<m_ndata; i++)
108  m_covariance[i][i] = pow(m_error[i], 2);
109 }
