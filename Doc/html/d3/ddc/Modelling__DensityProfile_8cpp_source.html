61 void cbl::modelling::densityprofile::Modelling_DensityProfile::set_model_DensityProfile_cosmology (const std::vector<cbl::cosmology::CosmologicalParameter> cosmo_param, const std::vector<statistics::PriorDistribution> cosmo_prior, const statistics::PriorDistribution concentration_prior, const statistics::PriorDistribution logM_prior, const statistics::PriorDistribution f_off_prior, const statistics::PriorDistribution sigma_off_prior)
62 {
63  m_data_model.Cpar = cosmo_param;
64 
65  const size_t nParams = cosmo_param.size()+4; // The total number of parameters is given by the cosmological ones + 4, since the density profile has 4 parameters (conc, logM, f_off, sigma_off)
66 
67  vector<statistics::ParameterType> Par_type (nParams, statistics::ParameterType::_Base_);
68  vector<string> Par_string (nParams);
69  std::vector<statistics::PriorDistribution> param_prior (nParams);
70 
71  // Set the names and priors of the cosmological parameters
72  for (size_t i=0; i<cosmo_param.size(); i++){
73  Par_string[i] = CosmologicalParameter_name(cosmo_param[i]);
74  param_prior[i] = cosmo_prior[i];
75  }
76 
77  // Set the names and priors for the density profile parameters
78  Par_string[cosmo_param.size()] = "concentration";
79  param_prior[cosmo_param.size()] = concentration_prior;
80  Par_string[cosmo_param.size()+1] = "logM";
81  param_prior[cosmo_param.size()+1] = logM_prior;
82  Par_string[cosmo_param.size()+2] = "f_off";
83  param_prior[cosmo_param.size()+2] = f_off_prior;
84  Par_string[cosmo_param.size()+3] = "sigma_off";
85  param_prior[cosmo_param.size()+3] = sigma_off_prior;
86 
87  // input data used to construct the model
88  auto inputs = make_shared<STR_Profile_data_model>(m_data_model);
89 
90  // set prior
91  m_set_prior(param_prior);
92 
93  // construct the model
94  if (m_2halo == false)
95  m_model = make_shared<statistics::Model1D>(statistics::Model1D(&nfw_1halo_allBins, nParams, Par_type, Par_string, inputs));
96  else
97  ErrorCBL("The 2-halo term is not implemented yet!", "set_model_DensityProfile_cosmology", "Modelling_DensityProfile.cpp");
98 }

103 double cbl::modelling::densityprofile::nfw_1halo (cbl::cosmology::Cosmology cosm, cbl::catalogue::Cluster cluster, const double radius, const double redshift, const double contrast, const double trunc_fact)
104 {
105  // Calculate the radius enclosing a given mass, in the cosmology considered
106  double H = cosm.HH(redshift)/cosm.HH(0.)*100./3.0857e+19; // in sec^-1
107  double rho_crit = 3*H*H/8/cbl::par::pi/6.6732e-8/1.98892e+33*3.0857e+24*3.0857e+24*3.0857e+24; // in h^2*M_sun/Mpc^3
108  double r_encl = pow( 3*cluster.mass()/(4*cbl::par::pi*contrast*rho_crit), 1./3. );
109 
110  double rscale = r_encl/cluster.concentration();
111  double trunc = trunc_fact*r_encl;
112  double tau = trunc/rscale;
113 
114  std::function<double(double)> F = [] (double x){
115  // both operations must be generalised to complex numbers when x<1
116  // in that case, both num and den are imaginary, and their ratio is real again
117  std::complex<double> num = std::acos(std::complex<double>(1./x,0.));
118  std::complex<double> den = std::sqrt(std::complex<double>(x*x-1.,0.));
119  std::complex<double> res = num/den;
120  return std::abs(real(res));
121  };
122  std::function<double(double)> L = [&tau] (double x){
123  return log(x/(sqrt(x*x+tau*tau)+tau));
124  };
125 
126  double rho_s = 1;
127  std::function<double(double)> integrate_mass = [&tau,&rscale,&rho_s] (double rad){
128  double x = rad/rscale;
129  double rho = rho_s/x/(1+x)/(1+x)*(tau*tau/(tau*tau+x*x))*(tau*tau/(tau*tau+x*x)); // in (M_sun/h)/((Mpc/h)^3)
130  return 4.*cbl::par::pi*rad*rad*rho;
131  };
132 
133  double a_meantau = 0, b_meantau = 0, inv_mis_scale = 1./cluster.sigma_off()/cluster.sigma_off();
134  std::function<double(double)> tau_factor1 = [&a_meantau,&b_meantau,&inv_mis_scale] (double s){
135  return (exp(-0.5*(s*s+a_meantau)*inv_mis_scale))/sqrt(b_meantau-a_meantau-s*s);
136  };
137  std::function<double(double)> tau_factor2 = [&a_meantau,&b_meantau,&inv_mis_scale] (double s){
138  return (exp(-0.5*(b_meantau-s*s)*inv_mis_scale))/sqrt(b_meantau-a_meantau-s*s);
139  };
140 
141 
142  // Define the variable tau_for_misc
143  double tau_for_misc = 0;
144 
145  std::function<double(double)> meantau_factor = [&tau_for_misc,&cluster,&a_meantau,&b_meantau,&inv_mis_scale,&tau_factor1,&tau_factor2] (double rad){
146  a_meantau = (tau_for_misc-rad)*(tau_for_misc-rad);
147  b_meantau = (tau_for_misc+rad)*(tau_for_misc+rad);
148  double c_meantau = (a_meantau+b_meantau)*0.5;
149  double upper_limit = std::min(sqrt(c_meantau-a_meantau),sqrt(25*cluster.sigma_off()*cluster.sigma_off()-a_meantau));
150  double meantau_factor = rad*2.*cbl::wrapper::gsl::GSL_integrate_cquad(tau_factor1, 0, upper_limit);
151  if (c_meantau < 25*cluster.sigma_off()*cluster.sigma_off())
152  {
153  double int_left = std::max(0.,sqrt(b_meantau-25*cluster.sigma_off()*cluster.sigma_off()));
154  double int_right = sqrt(b_meantau-c_meantau);
155  meantau_factor += rad*2.*cbl::wrapper::gsl::GSL_integrate_cquad(tau_factor2, int_left, int_right);
156  }
157  return meantau_factor;
158  };
159 
160  // Calculate rho_s given r_encl
161  double mass_int = cbl::wrapper::gsl::GSL_integrate_cquad(integrate_mass,0.,r_encl);
162  rho_s = cluster.mass()/mass_int; // scale density (=M0/(4 pi rs^3))
163 
164  std::function<double(double)> sigma_cen_fc = [&rscale,&tau,&rho_s,&F,&L] (double radius){
165  double x = radius/rscale;
166  double tausq = tau*tau; double tausq_xsq = tausq+x*x; double tau4th = tausq*tausq;
167  double prefact = rho_s*rscale*tau4th/(tausq+1.)/(tausq+1.)/(tausq+1.);
168  double a = 2*(tausq+1.)/(x*x-1.)*(1-F(x));
169  double b = 8*F(x);
170  double c = (tau4th-1.)/tausq/tausq_xsq;
171  double d = -cbl::par::pi*(4*tausq_xsq+tausq+1.)/pow(tausq_xsq,1.5);
172  double e = (tausq*(tau4th-1.)+(tausq_xsq)*(3*tau4th-6*tausq-1))*L(x)/(tau*tau*tau)/pow(tausq_xsq,1.5);
173  return prefact*(a+b+c+d+e)/1.e+12; // in (M_sun/h)/((pc/h)^2) = h*M_sun/pc^2
174  };
175  std::function<double(double)> sigma_cen_int = [&sigma_cen_fc] (double rad){
176  return sigma_cen_fc(rad)*2.*rad;
177  };
178  std::function<double(double)> meansigma_off_int = [&tau_for_misc,&radius,&cluster,&sigma_cen_fc,&meantau_factor] (double tau){
179  tau_for_misc = tau;
180  double int_left = std::max(0.,tau-5*cluster.sigma_off());
181  double int_right = std::min(radius,tau+5*cluster.sigma_off());
182  return tau * sigma_cen_fc(tau) * cbl::wrapper::gsl::GSL_integrate_cquad(meantau_factor, int_left, int_right);
183  };
184  std::function<double(double)> sigma_off_int = [&radius,&cluster,&a_meantau,&b_meantau,&inv_mis_scale,&tau_factor1,&tau_factor2,&sigma_cen_fc] (double tau){
185  a_meantau = (tau-radius)*(tau-radius);
186  b_meantau = (tau+radius)*(tau+radius);
187  double c_meantau = (a_meantau+b_meantau)*0.5;
188  double upper_limit = std::min(sqrt(c_meantau-a_meantau),sqrt(25*cluster.sigma_off()*cluster.sigma_off()-a_meantau));
189  double sigma_off_int = radius*2.*cbl::wrapper::gsl::GSL_integrate_cquad(tau_factor1, 0., upper_limit);
190  if (c_meantau < 25*cluster.sigma_off()*cluster.sigma_off())
191  {
192  double int_left = std::max(0.,sqrt(b_meantau-25*cluster.sigma_off()*cluster.sigma_off()));
193  double int_right = sqrt(b_meantau-c_meantau);
194  sigma_off_int += radius*2.*cbl::wrapper::gsl::GSL_integrate_cquad(tau_factor2, int_left, int_right);
195  }
196  return sigma_off_int * tau * sigma_cen_fc(tau);
197  };
198 
199  double sigma_cen = sigma_cen_fc(radius);
200  double meansigma_cen = cbl::wrapper::gsl::GSL_integrate_cquad(sigma_cen_int,0.,radius) /radius/radius;
201 
202  double deltasigma_cen = meansigma_cen - sigma_cen;
203 
204  double deltasigma_1h = 0, meansigma_off = 0, integrated_sigma_off = 0, deltasigma_off = 0;
205  if (cluster.f_off() <= 1.e-5 || cluster.sigma_off() <= 1.e-5){
206  deltasigma_1h = deltasigma_cen;
207  }else{
208  meansigma_off = cbl::wrapper::gsl::GSL_integrate_cquad(meansigma_off_int,0.,radius+5.*cluster.sigma_off()) *2./cbl::par::pi/cluster.sigma_off()/cluster.sigma_off()/radius/radius;
209  integrated_sigma_off = cbl::wrapper::gsl::GSL_integrate_cquad(sigma_off_int,std::max(0., radius-5*cluster.sigma_off()),radius+5.*cluster.sigma_off()) /radius/cbl::par::pi/cluster.sigma_off()/cluster.sigma_off();
210  deltasigma_off = meansigma_off - integrated_sigma_off;
211  deltasigma_1h = (1-cluster.f_off())*deltasigma_cen+cluster.f_off()*deltasigma_off;
212  }
213 
214  return deltasigma_1h;
215 }

219 std::vector<double> cbl::modelling::densityprofile::nfw_1halo_allBins (const std::vector<double> radius, const std::shared_ptr<void> inputs, std::vector<double> &parameter)
220 {
221  // structure contaning the required input data
222  shared_ptr<STR_Profile_data_model> pp = static_pointer_cast<STR_Profile_data_model>(inputs);
223 
224  // redefine the cosmology
225  cbl::cosmology::Cosmology cosmo = *pp->cosmology;
226 
227  // redefine the cluster object
228  cbl::catalogue::Cluster cluster = *pp->cluster;
229 
230  // set the cosmological parameters
231  for (size_t i=0; i<pp->Cpar.size(); ++i)
232  cosmo.set_parameter(pp->Cpar[i], parameter[i]);
233 
234  // set the cluster parameters
235  cluster.set_concentration(parameter[pp->Cpar.size()]);
236  cluster.set_mass(pow(pp->logM_base, parameter[pp->Cpar.size()+1])*1.e14);
237  cluster.set_f_off(parameter[pp->Cpar.size()+2]);
238  cluster.set_sigma_off(parameter[pp->Cpar.size()+3]);
239 
240  std::vector<double> density_profile(radius.size());
241 
242  for (size_t j=0; j<radius.size(); j++)
243  density_profile[j] = nfw_1halo(cosmo, cluster, radius[j], pp->redshift, pp->contrast, pp->trunc_fact);
244 
245  return density_profile;
246 }
