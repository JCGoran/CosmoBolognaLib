31 # construct the dataset by reading an input file
32 data = cbl.Data1D(dir_input+file_data)
33 
34 # set the stuff used to construct the model: here an object of class cosmology, just as an example
35 cosmology = cbl.Cosmology()
36 
37 # set the model to construct the likelihood
38 model = getModel1D(cosmology)
39 
40 # define a Gaussian likelihood
41 likelihood = cbl.Likelihood(data, model, cbl.LikelihoodType__Gaussian_Error_)
42 
43 # set the model parameter
44 valA, valB, valC = 1., 1., 1.
45 start = [valA, valB]
46 
47 # fix the parameter C, i.e. parameter[2], to valC=1
48 likelihood.parameters().fix(2, valC)
49 
50 # write the model
51 likelihood.write_model(dir_output, file_output_start, start)
52 
53 # limits for A and B
54 minA, maxA = -10., 10.
55 minB, maxB = -10., 10.
56 limits = [ [minA, maxA] , [minB, maxB] ]
57 
58 # maximize the likelihood and write the output
59 likelihood.maximize(start, limits)
60 likelihood.write_model_at_bestfit(dir_output, file_output_bestfit)
61 
62 # construct the priors
63 prior_A = cbl.PriorDistribution(cbl.DistributionType__Uniform_, minA, maxA)
64 prior_B = cbl.PriorDistribution(cbl.DistributionType__Uniform_, minB, maxB)
65 prior_C = cbl.PriorDistribution(cbl.DistributionType__Constant_, valC)
66 prior_distributions = cbl.PriorDistributionPtrVector([prior_A, prior_B, prior_C]) #
67 
68 # construct the posterior
69 posterior = cbl.Posterior(prior_distributions, likelihood, 696)
70 
71 # sample the posterior (starting the MCMC chain from the maximum of the posterior to speed up the chain convergence)
72 nwalkers = 10;
73 chain_size = 5000;
74 posterior.initialize_chains(chain_size, nwalkers, 1.e-5, [valA, valB])
75 posterior.sample_stretch_move(2)
76 
77 # show the median MCMC values of the four parameters on screen
78 print ("\n")

80  print("Posterior median of %s = %g\n"%(posterior.parameters().name(i), posterior.parameters().bestfit_value(i)))
