151 shared_ptr<Data> cbl::data::Data2D_extra::cut (const double xmin, const double xmax, const double ymin, const double ymax) const
152 {
153  vector<bool> mask(m_ndata, true);
154  vector<double> xx, yy;
155 
156  for (int i=0; i<m_xsize; i++)
157  for (int j=0; j<m_ysize; j++)
158  if (((m_x[i] < xmin) || (m_x[i]>xmax)) || ((m_y[j] < ymin) || (m_y[j]>ymax)))
159  mask[j+m_ysize*i] = false;
160 
161  for (int i=0; i<m_xsize; i++)
162  if (!((m_x[i] < xmin) || (m_x[i]>xmax)))
163  xx.push_back(m_x[i]);
164 
165  for (int i=0; i<m_ysize; i++)
166  if (!((m_y[i] < ymin) || (m_y[i]>ymax)))
167  yy.push_back(m_y[i]);
168 
169  vector<double> data, error;
170  vector<vector<double>> covariance;
171  vector<vector<double>> extra_info;
172 
173  int ndata_eff = 0;
174 
175  for (int i=0; i<m_ndata; i++)
176  if (mask[i])
177  ndata_eff ++;
178 
179  if (ndata_eff<1)
180  ErrorCBL("no elements left!", "cut", "Data2D_extra.cpp");
181 
182  data.resize(ndata_eff, 0);
183  error.resize(ndata_eff, 0);
184  covariance.resize(ndata_eff, vector<double>(ndata_eff, 0));
185  extra_info.resize(m_extra_info.size(), vector<double>(ndata_eff, 0));
186 
187  int index1 = 0;
188  for (int i=0; i<m_ndata; i++) {
189  if (mask[i]) {
190  data[index1] = m_data[i];
191  error[index1] = m_error[i];
192 
193  for (size_t j=0; j<m_extra_info.size(); j++) {
194  extra_info[j][index1] = m_extra_info[j][i];
195  }
196 
197  int index2 = 0;
198  for (int j=0; j<m_ndata; j++) {
199  if (mask[j]) {
200  covariance[index1][index2] = m_covariance[i][j];
201  index2++;
202  }
203  }
204  index1++;
205  }
206  }
207 
208  shared_ptr<Data> dd = make_shared<Data2D_extra>(Data2D_extra(xx, yy, data, covariance, extra_info));
209 
210  return dd;
211 }
