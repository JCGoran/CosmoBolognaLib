162 void cbl::statistics::Posterior::maximize (const vector<double> start, const unsigned int max_iter, const double tol, const double epsilon)
163 {
164  vector<double> starting_par;
165 
166  unsigned int npar_free = m_model->parameters()->nparameters_free();
167  unsigned int npar = m_model->parameters()->nparameters();
168 
169  if (start.size()==npar_free)
170  starting_par=start;
171  else if (start.size()==npar)
172  for(size_t i=0; i<npar_free; i++)
173  starting_par.push_back(start[m_model->parameters()->free_parameters()[i]]);
174  else
175  ErrorCBL("Error in maximize of Posterior.cpp: Check your inputs!");
176 
177  function<double(vector<double> &)> post = [this](vector<double> & pp) {
178  return -this->log(pp);
179  };
180 
181  vector<vector<double>> ranges;
182 
183  for (size_t i=0; i<npar_free; i++) {
184  int index = m_model->parameters()->free_parameters()[i];
185  ranges.push_back( {m_model->parameters()->prior_distribution(index)->xmin(), m_model->parameters()->prior_distribution(index)->xmax()} );
186  }
187 
188  coutCBL << "Maximizing the posterior..." << endl;
189  vector<double> result = cbl::gsl::GSL_minimize_nD(post, starting_par, ranges, max_iter, tol, epsilon);
190  coutCBL << "Done!" << endl << endl;
191 
192  m_model_parameters->set_bestfit_value(result);
193  m_model_parameters->write_bestfit_info();
194  coutCBL << "log(posterior) = " << post(result) << endl << endl;
195 }

285  const int nparameters = m_model_parameters->nparameters();

288  string MK = "mkdir -p "+output_dir;
289  if (system(MK.c_str())) {}
290 
291  vector<string> names;
292  names.push_back("Step");
293  for (int k=0; k<nparameters; k++)
294  names.emplace_back(m_model_parameters->name(k));
295 
296  names.push_back("Log(Likelihood)");
297  names.push_back("Log(Prior)");
298  names.push_back("Log(Posterior)");
299 
300 
301  vector<vector<double>> values(nparameters+4);
302 
303  int n=0;
304  for (int j=start; j<chain_size; j+=thin)
305  for (int i=0; i<nwalkers; i++) {
306  values[0].emplace_back(n);
307  vector<double> pp(nparameters);
308  for (int k=0; k<nparameters; k++) {
309  values[k+1].emplace_back(m_model_parameters->chain_value(k, j, i));
310  pp[k] = m_model_parameters->chain_value(k, j, i);
311  }
312  double lpr = m_prior->log(pp);
313  values[nparameters+1].emplace_back(m_logposterior_values[j*nwalkers+i]-lpr);
314  values[nparameters+2].emplace_back(lpr);
315  values[nparameters+3].emplace_back(m_logposterior_values[j*nwalkers+i]);
316  n++;
317  }
318 
319  cbl::ccfitswrapper::write_table_fits (output_dir, output_file, names, values);
320 
321 
322  coutCBL << "I wrote the file: " << output_dir+output_file << endl;
323 }

341 void cbl::statistics::Posterior::read_chain_ascii (const string input_dir, const string input_file, const int nwalkers, const int skip_header)
342 {
343  string file = input_dir+input_file;
344  coutCBL << "Reading the chain file " << file << endl;
345 
346  ifstream fin(file.c_str()); checkIO(fin, file);
347 
348  string line;
349  for(int i=0; i<skip_header; i++)
350  getline(fin, line);
351 
352  vector<vector<double>> chain_value;
353  m_logposterior_values.erase(m_logposterior_values.begin(), m_logposterior_values.end());
354 
355  while(getline(fin, line))
356  {
357  stringstream ss(line);
358  double NUM;
359  vector<double> ll, params;
360 
361  while(ss>>NUM) ll.push_back(NUM);
362  for(size_t i=1;i<ll.size()-3; i++){
363  params.push_back(ll[i]);
364  }
365 
366  m_logposterior_values.push_back(ll[ll.size()-1]);
367  chain_value.push_back(params);
368  }
369  int chain_size = chain_value.size()/nwalkers;
370 
371  fin.clear(); fin.close();
372 
373  checkDim(chain_value, nwalkers*chain_size, m_model_parameters->nparameters(), "chain_from_file");
374  checkDim(m_logposterior_values, nwalkers*chain_size, "logposterior_from_file");
375 
376  chain_value = cbl::transpose(chain_value);
377 
378  m_model_parameters->set_chain_values(chain_value, nwalkers);
379 
380  coutCBL << "Done!" << endl << endl;
381 }
