74 void cbl::statistics::PosteriorParameters::m_set_parameter_type ()
75 {
76  m_nparameters_free = 0;
77  m_nparameters_fixed = 0;
78  m_nparameters_base = 0;
79  m_nparameters_derived = 0;
80 
81  m_base_parameters.erase(m_base_parameters.begin(), m_base_parameters.end());
82  m_fixed_parameters.erase(m_fixed_parameters.begin(), m_fixed_parameters.end());
83  m_free_parameters.erase(m_free_parameters.begin(), m_free_parameters.end());
84  m_derived_parameters.erase(m_derived_parameters.begin(), m_derived_parameters.end());
85 
86  for (size_t i=0; i<m_nparameters; i++) {
87  switch (m_parameter_type[i]) {
88  case statistics::ParameterType::_Base_:
89  if(m_parameter_prior[i] != NULL){
90  if (m_parameter_prior[i]->distributionType()==glob::DistributionType::_Constant_) {
91  m_nparameters_fixed +=1;
92  m_fixed_parameters.push_back(i);
93  }
94  else {
95  m_nparameters_free +=1;
96  m_free_parameters.push_back(i);
97  }
98  m_nparameters_base +=1;
99  m_base_parameters.push_back(i);
100  }
101  break;
102 
103  case statistics::ParameterType::_Derived_:
104  m_nparameters_derived += 1;
105  m_derived_parameters.push_back(i);
106  break;
107 
108  default:
109  ErrorCBL("Error in cbl::statistics::PosteriorParameters of PosteriorParameters.cpp: no such kind of parameter!");
110  }
111  }
