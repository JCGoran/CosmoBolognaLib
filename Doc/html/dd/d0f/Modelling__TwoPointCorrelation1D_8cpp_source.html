67 void cbl::modelling::twopt::Modelling_TwoPointCorrelation1D::set_data_model (const cosmology::Cosmology cosmology, const double redshift, const string method_Pk, const double sigmaNL_perp, const double sigmaNL_par, const bool NL, const double bias, const double pimax, const double r_min, const double r_max, const double k_min, const double k_max, const int step, const string output_dir, const string output_root, const int norm, const double aa, const bool GSL, const double prec, const string file_par, const double Delta, const bool isDelta_vir, const std::vector<double> cluster_redshift, const std::vector<double> cluster_mass_proxy, const std::vector<double> cluster_mass_proxy_error, const string model_bias, const string meanType, const int seed, const cosmology::Cosmology cosmology_mass, const std::vector<double> redshift_source)

80  m_data_model->r_max = r_max;
81  m_data_model->k_min = k_min;
82  m_data_model->k_max = k_max;
83  m_data_model->step = step;
84  m_data_model->output_dir = output_dir;
85  m_data_model->output_root = output_root;
86  m_data_model->norm = norm;
87  m_data_model->aa = aa;
88  m_data_model->GSL = GSL;
89  m_data_model->prec = prec;
90  m_data_model->file_par = file_par;
91  m_data_model->bias = bias;
92  m_data_model->Delta = (isDelta_vir) ? m_data_model->cosmology->Delta_vir(Delta, redshift) : Delta;
93  m_data_model->model_bias = model_bias;
94  m_data_model->meanType = meanType;
95  m_data_model->gau_ran = make_shared<random::NormalRandomNumbers>(random::NormalRandomNumbers(0., 1., seed));
96 
97  if (m_data_model->cosmology->sigma8()>0) {
98  m_data_model->sigma8_z = m_data_model->cosmology->sigma8(m_data_model->redshift);
99  }
100  else {
101  coutCBL << "sigma8 is not set, it will be computed from the power spectrum with " << m_data_model->method_Pk << endl;
102  m_data_model->sigma8_z = m_data_model->cosmology->sigma8_Pk(m_data_model->method_Pk, m_data_model->redshift, m_data_model->output_root);
103  coutCBL << "--> sigma8(z=" << m_data_model->redshift << ") = " << m_data_model->sigma8_z << endl << endl;
104  }
105  m_data_model->linear_growth_rate_z = m_data_model->cosmology->linear_growth_rate(m_data_model->redshift, 1.);
106 
107 
108  m_data_model->DVfid = m_data_model->cosmology->D_V(m_data_model->redshift);
109  m_data_model->DAfid = m_data_model->cosmology->D_A(m_data_model->redshift);
110  m_data_model->HHfid = m_data_model->cosmology->HH(m_data_model->redshift);
111 
112  if (cluster_mass_proxy.size()>0) {
113  vector<double> cl_zz(cluster_mass_proxy.size(), m_data_model->redshift), cl_sigma(cluster_mass_proxy.size());
114  if (cluster_redshift.size()==cluster_mass_proxy.size())
115  cl_zz = cluster_redshift;
116  for (size_t i=0; i<cl_sigma.size(); ++i)
117  cl_sigma[i] = sqrt(m_data_model->cosmology->sigma2M(cluster_mass_proxy[i], m_data_model->method_Pk, 0., m_data_model->output_root, "Spline", m_data_model->k_max));
118 
119  m_data_model->cluster_mass_proxy = make_shared<data::Data1D_extra>(data::Data1D_extra(cl_zz, cluster_mass_proxy, cluster_mass_proxy_error, {cl_sigma}));
120 
121  m_data_model->cosmology_mass = cosmology_mass;
122  m_data_model->redshift_source = redshift_source;
123  }
124 }
