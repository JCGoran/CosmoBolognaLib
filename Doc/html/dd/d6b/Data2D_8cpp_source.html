294 shared_ptr<Data> cbl::data::Data2D::cut(const double xmin, const double xmax, const double ymin, const double ymax) const
295 {
296  vector<bool> mask(m_ndata, true);
297  vector<double> xx, yy;
298 
299  for (int i=0; i<m_xsize; i++) {
300  for (int j=0; j<m_ysize; j++) {
301  if ( ((m_x[i] < xmin) || (m_x[i]>xmax)) || ((m_y[j] < ymin) || (m_y[j]>ymax)))
302  mask[j+m_ysize*i] = false;
303  }
304  }
305 
306  for (int i=0; i<m_xsize; i++)
307  if ( !((m_x[i] < xmin) || (m_x[i]>xmax)))
308  xx.push_back(m_x[i]);
309 
310  for (int i=0; i<m_ysize; i++)
311  if (!((m_y[i] < ymin) || (m_y[i]>ymax)))
312  yy.push_back(m_y[i]);
313 
314  vector<double> data, error;
315  vector<vector<double>> covariance;
316  Data::cut(mask, data, error, covariance);
317 
318  shared_ptr<Data> dd = make_shared<Data2D>(Data2D(xx, yy, data, covariance));
