100 void cbl::statistics::SuperSampleCovariance::m_compute_topHat_window (const double delta_z, const std::vector<double> redshift_edges)
101 {
102  m_nsteps = (int)((redshift_edges[redshift_edges.size()-1]-redshift_edges[0])/delta_z + 1);
103  if (m_nsteps < m_nbins)
104  cbl::ErrorCBL("m_nsteps can not be lower than the number of redshift bins!","m_compute_topHat_window","SuperSampleCovariance");
105 
106  m_redshifts.resize(m_nsteps);
107  m_windows.resize(m_nbins);
108 
109  for (int i=0; i<m_nbins; i++) {
110  m_windows[i].resize(m_nsteps);
111  double iter = redshift_edges[0]-delta_z;
112  double Delta_z = redshift_edges[i+1] - redshift_edges[i];
113  for (int j=0; j<m_nsteps; j++) {
114  iter += delta_z;
115  m_redshifts[j] = iter;
116  if ((iter>redshift_edges[i]) && (iter<=redshift_edges[i+1]))
117  m_windows[i][j] = 1/Delta_z;
118  }
119  }
120 }

126 void cbl::statistics::SuperSampleCovariance::m_compute_gaussian_window (const double delta_z, const std::vector<double> W_mean, const std::vector<double> W_std)
127 {
128  double max_z = W_mean[W_mean.size()-1]+3.5*W_std[W_mean.size()-1];
129  double min_z = W_mean[0]-3.5*W_std[0];
130 
131  m_nsteps = (int)((max_z - min_z) / delta_z + 1);
132  m_redshifts.resize(m_nsteps);
133  m_windows.resize(m_nbins);
134 
135  for (int i=0; i<m_nbins; i++) {
136  m_windows[i].resize(m_nsteps);
137  double iter = min_z - delta_z;
138  for (int j=0; j<m_nsteps; j++) {
139  iter += delta_z;
140  m_redshifts[j] = iter;
141  m_windows[i][j] = exp( - (m_redshifts[j] - W_mean[i]) * (m_redshifts[j] - W_mean[i]) / (2*W_std[i]*W_std[i]) ) / sqrt(2*cbl::par::pi*W_std[i]*W_std[i]);
142  }
143  }
144 }

150 std::vector<std::vector<double>> cbl::statistics::SuperSampleCovariance::compute_Sij (cbl::cosmology::Cosmology cosmo) const
151 {
152  // Compute comoving distances, volumes, growth factor
153  std::vector<double> comov_dist(m_nsteps), dV(m_nsteps), growthf(m_nsteps);
154  for (int i=0; i<m_nsteps; i++){
155  comov_dist[i] = cosmo.D_C(m_redshifts[i]);
156  dV[i] = comov_dist[i] * comov_dist[i] * cosmo.D_H() * cosmo.EE_inv(m_redshifts[i]); // D_H()*EE_inv(z) is the D_C normalized derivative
157  growthf[i] = cosmo.gg(m_redshifts[i])/cosmo.gg(0)/(1+m_redshifts[i]); // normalized scale-independent growth factor
158  }
159 
160  // Compute normalizations
161  std::vector<double> Inorm(m_nbins);
162  std::vector<std::vector<double>> integrand (m_nbins,std::vector<double>(m_nsteps));
163  for (int i=0; i<m_nbins; i++){
164  for (int s=0; s<m_nsteps; s++){
165  integrand[i][s] = dV[s] * m_windows[i][s] * m_windows[i][s] / 1.e10;
166  }
167  cbl::glob::FuncGrid integ(m_redshifts, integrand[i], "Spline");
168  Inorm[i] = integ.integrate_cquad(m_redshifts[0], m_redshifts[m_redshifts.size()-1]) * 1.e10;
169  }
170 
171  // Compute U(k)
172  const double h = cosmo.hh();
173  const double keq = 0.02/h; // Equality matter radiation in 1/Mpc (more or less)
174  const double klogwidth = 10; // Factor of width of the integration range. 10 seems ok
175  double max_comov_dist = comov_dist[comov_dist.size()-1];
176  double min_comov_dist = comov_dist[0];
177 
178  double kmin = 0;
179  double kmax = 0;
180  if (keq<(1./max_comov_dist))
181  kmin = keq/klogwidth;
182  else
183  kmin = (1./max_comov_dist)/klogwidth;
184  if (keq>(1./min_comov_dist))
185  kmax = keq*klogwidth;
186  else
187  kmax = min_comov_dist*klogwidth;
188 
189  const double nk = pow(2,m_precision); // m_precision=10 seems to be enough. Increase to test precision, reduce to speed up.
190  const double logkmin = log(kmin);
191  const double logkmax = log(kmax);
192  std::vector<double> logk(nk);
193  std::vector<double> kk(nk);
194  for (size_t i=0; i<logk.size(); i++) {
195  logk[i]=logkmin+i*(logkmax-logkmin)/(nk-1);
196  kk[i]=exp(logk[i]);
197  }
198 
199  std::vector<double> Pk_new = cosmo.Pk_DM(kk, m_method_Pk, m_NL, 0., m_store_output, "test", -1, 0.0001, 100, 1.e-2, cbl::par::defaultString, false);
200  std::vector<std::vector<double>> Uarr(m_nbins, std::vector<double>(logk.size()));
201  std::vector<double> kr(m_nsteps);
202  std::vector<std::vector<double>> integrand2(m_nbins, std::vector<double>(m_nsteps));
203  for (int i=0; i<m_nbins; i++) {
204  for (size_t j=0; j<logk.size(); j++) {
205  for (size_t s=0; s<m_redshifts.size(); s++) {
206  kr[s] = kk[j]*comov_dist[s];
207  integrand2[i][s] = dV[s] * m_windows[i][s] * m_windows[i][s] * growthf[s] * sin(kr[s]) / kr[s] / 1.e10;
208  }
209  cbl::glob::FuncGrid integ(m_redshifts, integrand2[i], "Spline");
210  Uarr[i][j] = integ.integrate_cquad(m_redshifts[0], m_redshifts[m_redshifts.size()-1]) * 1.e10;
211  }
212  }
213 
214  // Compute S_ij
215  std::vector<std::vector<double>> Cl_zero(m_nbins,std::vector<double>(m_nbins));
216  std::vector<std::vector<double>> U1(m_nbins, std::vector<double>(logk.size()));
217  std::vector<std::vector<double>> U2(m_nbins, std::vector<double>(logk.size()));
218  std::vector<std::vector<double>> integrand3(m_nbins, std::vector<double>(logk.size()));
219  for (int i=0; i<m_nbins; i++){
220  for (size_t j=0; j<logk.size(); j++){
221  U1[i][j] = Uarr[i][j]/Inorm[i];
222  }
223  for (int k=i; k<m_nbins; k++){
224  for (size_t j=0; j<logk.size(); j++){
225  U2[k][j] = Uarr[k][j]/Inorm[k];
226  integrand3[k][j] = kk[j] * kk[j] * Pk_new[j] * U1[i][j] * U2[k][j] * 1.e10;
227  }
228  cbl::glob::FuncGrid integ(kk, integrand3[k], "Spline");
229  Cl_zero[i][k] = 2/cbl::par::pi * integ.integrate_cquad(kk[0], kk[kk.size()-1]) / 1.e10;
230  }
231  }
232 
233  // Fill by symmetry
234  for (int i=0; i<m_nbins; i++){
235  for (int j=0; j<m_nbins; j++){
236  Cl_zero[i][j] = Cl_zero[std::min(i,j)][std::max(i,j)];
237  }
238  }
239  std::vector<std::vector<double>> Sij(m_nbins, std::vector<double>(m_nbins));
240  for (int i=0; i<m_nbins; i++){
241  for (int j=0; j<m_nbins; j++){
242  Sij[i][j] = Cl_zero[i][j] / m_area; // S_ij is given by dividing Cl_zero/4pi by the fraction of the sky, i.e. Area(steradians)/4pi
243  }
244  }
245 
246  return Sij;
247 }
