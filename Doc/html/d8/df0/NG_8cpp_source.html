85  pow(kk[0]*kk[1],m_n_spec-4.0)-pow(kk[0]*kk[2],m_n_spec-4.0)-pow(kk[1]*kk[2],m_n_spec-4.0)-2.0*pow(kk[0]*kk[1]*kk[2],2.0*(m_n_spec-4.0)/3.0);
86  bs *= 6.0*gsl_pow_2(bm);
87  break;
88 
89  case 3: // Enfolded shape
90  bs = pow(kk[0]*kk[1],m_n_spec-4.0)+pow(kk[0]*kk[2],m_n_spec-4.0)+pow(kk[1]*kk[2],m_n_spec-4.0)+3.0*pow(kk[0]*kk[1]*kk[2],2.0*(m_n_spec-4.0)/3.0)-

100  3.0*pow(kk[2],(m_n_spec-4.0)/3.0)*pow(kk[1],2.0*(m_n_spec-4.0)/3.0)*pow(kk[0],m_n_spec-4.0)+3.0*pow(kk[0],(m_n_spec-4.0)/3.0)*pow(kk[2],2.0*(m_n_spec-4.0)/3.0)*pow(kk[1],m_n_spec-4.0)-

116 double cbl::cosmology::Cosmology::mrk (const double kk, const double mass, const string method_Pk, const string output_root, const int norm, const double k_min, const double k_max, const double prec, const string file_par)
117 {
118  double xx = kk * Radius(mass, m_RhoZero);
119 
120  double AA = Am(method_Pk, output_root, norm, k_min, k_max, prec, file_par);
121 
122  double TT = sqrt(Pk(kk, method_Pk, false, 0., output_root, norm, k_min, k_max, prec, file_par) / AA / pow(kk,m_n_spec));
123 
124  return 5.99170e6 * gsl_pow_2(kk) * TopHat_WF(xx) * TT / m_Omega_matter;
125 }

132 double cbl::cosmology::Cosmology::bias_kernel (double xx, void *params)
133 {
134  cbl::glob::GSL_f_pars *pp = (cbl::glob::GSL_f_pars *)params;
135 
136  int ni = 16;
137  double *XX = new double[ni];
138  double *Weight = new double[ni];
139  gauleg (0., 1., XX, Weight, ni);
140 
141  vector<double> km(3);
142  km[0] = xx;
143  km[2] = pp->kt;
144 
145  double mass = pp->mass;
146  string method_Pk = pp->method_Pk;
147  string output_root = pp->output_root;
148  int norm = pp->norm;
149  double k_min = pp->k_min;
150  double k_max = pp->k_max;
151  double prec = pp->prec;
152  string file_par = pp->file_par;
153 
154 
155  const double xi1 = -1.0, xi2 = 1.0;
156  double yi = 0.0;
157 
158  for (int i=0; i<ni; i++) {
159  double xi = xi1+(xi2-xi1)*XX[i];
160  km[1] = sqrt(gsl_pow_2(xx)+gsl_pow_2(pp->kt)+2.0*xx*pp->kt*xi);
161  yi += bispectrum(km, method_Pk, output_root, norm, k_min, k_max, prec, file_par)*mrk(km[1], mass, method_Pk, output_root, norm, k_min, k_max, prec, file_par)*Weight[i];

165 
166  return yi;
167 }

502 double cbl::cosmology::Cosmology::dskewnessdM (const double mass, const string method_Pk, const string output_root, const string interpType, const int norm, const double k_min, const double k_max, const double prec, const string input_file, const bool is_parameter_file)

545 double cbl::cosmology::Cosmology::MF_correction (const double mass, const double redshift, const string method_Pk, const string output_root, const string interpType, const int norm, const double k_min, const double k_max, const double prec, const string input_file, const bool is_parameter_file)
546 {
547  double dc = deltac(redshift)*sqrt(0.8);
548  double gf = DD(redshift);
549 
550  double SSS = sigma2M(mass, method_Pk, 0., output_root, interpType, k_max, input_file, is_parameter_file);

552 
553  double DlnSigmaDlnM = dnsigma2M(1, mass, method_Pk, 0., output_root, interpType, k_max, input_file, is_parameter_file)*(mass/(2.*SSS));
554

558

561  double mfc = mfc1+mfc2;
562  double psg = 1.;
563 
564  if (mfc > 0.0) mfc = 0.0;
565  else psg = DlnSigmaDlnM*(dc/gf/sm)/mass;
566
