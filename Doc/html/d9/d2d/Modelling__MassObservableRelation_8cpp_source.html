298 double cbl::modelling::massobsrel::mass_from_counts (double (*fz)(std::vector<double>, std::shared_ptr<void>), double (*z_error)(std::vector<double>), double (*proxy_error)(std::vector<double>), const double redshift_min, const double redshift_max, const double proxy_min, const double proxy_max, cbl::cosmology::Cosmology cosmology, cbl::catalogue::Cluster cluster, const double Area, const std::string model_MF, const bool store_output, const double Delta, const bool isDelta_vir, const cbl::glob::FuncGrid interp_sigmaM, const cbl::glob::FuncGrid interp_DlnsigmaM, const double proxy_pivot, const double z_pivot, const double mass_pivot, const double log_base)
299 {
300  double fact = (cosmology.unit()) ? 1 : cosmology.hh();
301  std::shared_ptr<void> pp;
302  auto cosmology_ptr = std::make_shared<cbl::cosmology::Cosmology>(cosmology);
303 
304  double the_minM = 0;
305 
306  // M_tot integrand
307  auto integrand_Mtot = [&the_minM,&z_error,&proxy_error,&fz,&cosmology_ptr,&cosmology,&cluster,&log_base,&proxy_pivot,&z_pivot,&mass_pivot,&proxy_min,&proxy_max,&redshift_min,&redshift_max,&interp_sigmaM,&interp_DlnsigmaM,&store_output,&fact,&model_MF,&Delta,&Area,&isDelta_vir,&pp] (const std::vector<double> x)
308  {
309  double Delta_ = (isDelta_vir) ? cosmology.Delta_vir(Delta, x[1]) : Delta;
310  double Mass = x[0]*mass_pivot*pow(log_base,the_minM);
311  double normM = x[0]*pow(log_base,the_minM);
312 
313  // Compute P(lambda|M,z)
314  double log_M = log(normM)/log(log_base);
315  double log_f_z = log( fz({x[1], z_pivot}, cosmology_ptr) )/log(log_base);
316 
317  double mean = cluster.alpha_scaling_rel() + cluster.beta_scaling_rel()*log_M + cluster.gamma_scaling_rel()*log_f_z + log(proxy_pivot)/log(log_base);
318  double sigma = cluster.scatter0_scaling_rel() + cluster.scatterM_scaling_rel()*pow(log_M, cluster.scatterM_exponent_scaling_rel()) + cluster.scatterz_scaling_rel()*pow(log_f_z, cluster.scatterz_exponent_scaling_rel());
319  double P_lambda__M_z = (cbl::gaussian(log(x[2])/log(log_base), pp, {mean,sigma})) / (x[2]*log(log_base));
320 
321  // Compute the integrals of P(z|z) and P(lambda|lambda)
322  double mean_Pz = x[1] + cluster.zbias() * (1+x[1]);
323  double int_P_z = 0.5 * ( erf( (redshift_max - mean_Pz) / (sqrt(2)*z_error({cluster.zerror(), redshift_max})) ) - erf( (redshift_min - mean_Pz) / (sqrt(2)*z_error({cluster.zerror(), redshift_min})) ) );
324  double int_P_lambda = 0.5 * ( erf( (proxy_max - x[2]) / (sqrt(2)*proxy_error({cluster.proxyerror(), proxy_max})) ) - erf( (proxy_min - x[2]) / (sqrt(2)*proxy_error({cluster.proxyerror(), proxy_min})) ) );
325 
326  return normM * cosmology.mass_function(Mass, interp_sigmaM(Mass*fact), interp_DlnsigmaM(Mass*fact), x[1], model_MF, store_output, cbl::par::defaultString, Delta_)*Area*cosmology.dV_dZdOmega(x[1], true) * P_lambda__M_z * int_P_z * int_P_lambda;
327  };
328 
329  // Pure counts integrand
330  auto integrand_counts = [&the_minM,&z_error,&proxy_error,&fz,&cosmology_ptr,&cosmology,&cluster,&log_base,&proxy_pivot,&z_pivot,&mass_pivot,&proxy_min,&proxy_max,&redshift_min,&redshift_max,&interp_sigmaM,&interp_DlnsigmaM,&store_output,&fact,&model_MF,&Delta,&Area,&isDelta_vir,&pp] (const std::vector<double> x)
331  {
332  double Delta_ = (isDelta_vir) ? cosmology.Delta_vir(Delta, x[1]) : Delta;
333  double Mass = x[0]*mass_pivot*pow(log_base,the_minM);
334  double normM = x[0]*pow(log_base,the_minM);
335 
336  // Compute P(lambda|M,z)
337  double log_M = log(normM)/log(log_base);
338  double log_f_z = log( fz({x[1], z_pivot}, cosmology_ptr) )/log(log_base);
339 
340  double mean = cluster.alpha_scaling_rel() + cluster.beta_scaling_rel()*log_M + cluster.gamma_scaling_rel()*log_f_z + log(proxy_pivot)/log(log_base);
341  double sigma = cluster.scatter0_scaling_rel() + cluster.scatterM_scaling_rel()*pow(log_M, cluster.scatterM_exponent_scaling_rel()) + cluster.scatterz_scaling_rel()*pow(log_f_z, cluster.scatterz_exponent_scaling_rel());
342  double P_lambda__M_z = (cbl::gaussian(log(x[2])/log(log_base), pp, {mean,sigma})) / (x[2]*log(log_base));
343 
344  // Compute the integrals of P(z|z) and P(lambda|lambda)
345  double mean_Pz = x[1] + cluster.zbias() * (1+x[1]);
346  double int_P_z = 0.5 * ( erf( (redshift_max - mean_Pz) / (sqrt(2)*z_error({cluster.zerror(), redshift_max})) ) - erf( (redshift_min - mean_Pz) / (sqrt(2)*z_error({cluster.zerror(), redshift_min})) ) );
347  double int_P_lambda = 0.5 * ( erf( (proxy_max - x[2]) / (sqrt(2)*proxy_error({cluster.proxyerror(), proxy_max})) ) - erf( (proxy_min - x[2]) / (sqrt(2)*proxy_error({cluster.proxyerror(), proxy_min})) ) );
348 
349  return cosmology.mass_function(Mass, interp_sigmaM(Mass*fact), interp_DlnsigmaM(Mass*fact), x[1], model_MF, store_output, cbl::par::defaultString, Delta_)*Area*cosmology.dV_dZdOmega(x[1], true) * P_lambda__M_z * int_P_z * int_P_lambda;
350  };
351 
352  // -------------------------------------------------------------
353 
354  // Find the minimum and maximum masses, given the parameters of the scaling relation
355  double scatter = cluster.scatter0_scaling_rel();
356 
357  double log_lambda_min = std::max(log(proxy_min/proxy_pivot)/log(log_base) - 3.5*scatter, log(0.0001/proxy_pivot)/log(log_base));
358  double log_lambda_max = log(proxy_max/proxy_pivot)/log(log_base) + 3.5*scatter;
359  double log_f_z_min = log( fz({redshift_min, z_pivot}, cosmology_ptr) )/log(log_base);
360  double log_f_z_max = log( fz({redshift_max, z_pivot}, cosmology_ptr) )/log(log_base);
361 
362  double M1 = (- cluster.alpha_scaling_rel() + log_lambda_min - cluster.gamma_scaling_rel()*log_f_z_min) / cluster.beta_scaling_rel();
363  double M2 = (- cluster.alpha_scaling_rel() + log_lambda_max - cluster.gamma_scaling_rel()*log_f_z_min) / cluster.beta_scaling_rel();
364  double M3 = (- cluster.alpha_scaling_rel() + log_lambda_min - cluster.gamma_scaling_rel()*log_f_z_max) / cluster.beta_scaling_rel();
365  double M4 = (- cluster.alpha_scaling_rel() + log_lambda_max - cluster.gamma_scaling_rel()*log_f_z_max) / cluster.beta_scaling_rel();
366 
367  double min1 = std::min(M1, M2);
368  double min2 = std::min(min1, M3);
369  double minM_ = std::min(min2, M4);
370  double max1 = std::max(M1, M2);
371  double max2 = std::max(max1, M3);
372  double maxM_ = std::max(max2, M4);
373 
374  double minM = std::max(minM_, log(1.e12/mass_pivot)/log(log_base));
375  double maxM = std::min(maxM_, log(1.e16/mass_pivot)/log(log_base));
376 
377  the_minM = minM;
378 
379 
380  // Define the integral limits
381  int integral_dimension=3;
382  std::vector<std::vector<double>> integration_limits(integral_dimension);
383  integration_limits[0] = {pow(log_base,minM)/pow(log_base,minM), pow(log_base,maxM)/pow(log_base,minM)};
384  integration_limits[1] = {std::max(redshift_min - 3.5*z_error({cluster.zerror(), redshift_min}), 0.), redshift_max + 3.5*z_error({cluster.zerror(), redshift_max})};
385  integration_limits[2] = {std::max(proxy_min - 3.5*proxy_error({cluster.proxyerror(), proxy_min}), 1.), proxy_max + 3.5*proxy_error({cluster.proxyerror(), proxy_max})};
386 
387  // Compute the M_tot integral
388  cbl::wrapper::cuba::CUBAwrapper CW (integrand_Mtot, integral_dimension);
389  double nc = CW.IntegrateVegas(integration_limits,false);
390 
391  // Divide it by the counts
392  cbl::wrapper::cuba::CUBAwrapper CW2 (integrand_counts, integral_dimension);
393  double counts = CW2.IntegrateVegas(integration_limits,false);
394 
395  if (integration_limits[0][0] < integration_limits[0][1] && counts > 0) {
396  nc /= counts;
397  return log(nc)/log(log_base);
398  } else
399  return cbl::par::defaultDouble;
400 }
