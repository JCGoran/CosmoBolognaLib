339 double cbl::gsl::GSL_root_brent (gsl_function Func, const double low_guess, const double up_guess, const double rel_err, const double abs_err)
340 {
341  gsl_set_error_handler_off();
342 
343  int status = 0, iter = 0, max_iter = 10000;
344  const gsl_root_fsolver_type *T;
345  double r;
346 
347  gsl_root_fsolver *s;
348  T = gsl_root_fsolver_brent;
349  s = gsl_root_fsolver_alloc(T);
350 
351  double x_lo = low_guess;
352  double x_hi = up_guess;
353 
354  gsl_root_fsolver_set(s, &Func, x_lo, x_hi);
355 
356  do
357  {
358  iter ++;
359  status = gsl_root_fsolver_iterate(s);
360 
361  if ((status!=GSL_SUCCESS) && (status!=GSL_CONTINUE))
362  check_GSL_fail(status, true, "GSL_root_brent", "gsl_root_fsolver_iterate");
363 
364  r = gsl_root_fsolver_root(s);
365  x_lo = gsl_root_fsolver_x_lower(s);
366  x_hi = gsl_root_fsolver_x_upper(s);
367 
368  status = gsl_root_test_interval(x_lo, x_hi, abs_err, rel_err);
369 
370  if ((status!=GSL_SUCCESS) && (status!=GSL_CONTINUE))
371  check_GSL_fail(status, true, "GSL_root_brent", "gsl_root_test_interval");
372  }
373 
374  while (status==GSL_CONTINUE && iter<max_iter);
375 
376  gsl_root_fsolver_free(s);
377 
378  check_GSL_fail(status, true, "GSL_minimize_nD", par::defaultString);
379 
380  return r;
