107 double cbl::glob::spherical_harmonics_coeff::power (const int l, const int bin1, const int bin2)
108 {
109  const int min_n = l*(l+1)/2;
110  double power = (real(min_n, bin1)*real(min_n, bin2)+imag(min_n, bin1)*imag(min_n, bin2));
111  for (int m=1; m<l+1; m++) {
112  const int pos = min_n+m;
113  power += 2.*(real(pos, bin1)*real(pos, bin2)+imag(pos, bin1)*imag(pos, bin2));
114  }
115 
116  return 4.*par::pi/(2*l+1)*power;
117 }

123 vector<double> cbl::glob::count_triplets_SphericalHarmonics (const double r12_min, const double r12_max, const double r13_min, const double r13_max, const int norders, const cbl::catalogue::Catalogue catalogue)
124 {
125  vector<double> zeta_l(norders, 0);
126 
127  std::shared_ptr<catalogue::Catalogue> cc(new catalogue::Catalogue(catalogue));
128 
129  chainmesh::ChainMesh_Catalogue cm;
130 
131  cm.set_par(r13_max*0.5, cc, r13_max);
132 
133  auto cat = cm.catalogue();
134 
135  const int nObjects = catalogue.nObjects();
136 
137  // start the multithreading parallelization
138 #pragma omp parallel num_threads(omp_get_max_threads())
139  {
140  spherical_harmonics_coeff alm(norders, 2);
141  vector<double> _zeta_l(norders, 0);
142 
143  // parallelized loop
144 #pragma omp for schedule(static, 2)
145  for (int i=0; i<nObjects; i++)
146  {
147  alm.reset();
148 
149  double ixx = cat->xx(i);
150  double iyy = cat->yy(i);
151  double izz = cat->zz(i);
152  double iww = cat->weight(i);
153 
154  vector<vector<double>> r1, r2;
155 
156  vector<long> close_objects = cm.close_objects({ixx, iyy, izz}, -1);
157 
158  for (size_t j=0; j<close_objects.size(); j++) {
159  double jxx = cat->xx(close_objects[j]);
160  double jyy = cat->yy(close_objects[j]);
161  double jzz = cat->zz(close_objects[j]);
162 
163  double xx = jxx-ixx;
164  double yy = jyy-iyy;
165  double zz = jzz-izz;
166 
167  double rr = sqrt(xx*xx+yy*yy+zz*zz);
168 
169  if(rr>0) {
170 
171  if (rr>=r12_min && rr<=r12_max)
172  alm.add (xx/rr, yy/rr, zz/rr, cat->weight(close_objects[j]), 0);
173 
174  if (rr>=r13_min && rr<=r13_max)
175  alm.add (xx/rr, yy/rr, zz/rr, cat->weight(close_objects[j]), 1);
176 
177  }
178  }
179 
180  for (int l=0; l<norders; l++)
181  _zeta_l[l] += iww*alm.power(l, 0, 1);
182 
183  }
184 
185 #pragma omp critical
186  {
187  for (int l=0; l<norders; l++)
188  zeta_l[l] += _zeta_l[l];
189  }
190 
191  }
192 
193  return zeta_l;
194 }

273  pairs[b1] += _pairs[b1];
274  for (int b2=0; b2<nbins; b2++)
275  for (int l=0; l<norders; l++)
276  triplets[b1][b2][l] += _triplets[b1][b2][l];
277  }
278  }
279 
280  }
281 }

340 {
341 
342  // ---------------------------------------------------------------
343  // ---------------- construct the mixed catalogue ----------------
344  // ---------------------------------------------------------------
345 
346  double N_R = double(random_catalogue.nObjects())/catalogue.nObjects();
347 
348  catalogue::Catalogue ran_catalogue (random_catalogue);
349  catalogue::Catalogue mixed_catalogue (catalogue);
350

352  double xx = random_catalogue.xx(i);
353  double yy = random_catalogue.yy(i);
354  double zz = random_catalogue.zz(i);
355  double ww = -random_catalogue.weight(i)/N_R;
356  ran_catalogue.set_var(i, catalogue::Var::_Weight_, -ww);
357 
358  auto obj = make_shared<catalogue::Object>( catalogue::Object());
359  obj->set_xx(xx);
360  obj->set_yy(yy);
361  obj->set_zz(zz);
362  obj->set_weight(ww);
363  mixed_catalogue.add_object(obj);
364  }
365 
366  // ------------------------------------------------------------------------------------
367  // ---------------- measure triplet multipoles expansion for R and D-R ----------------

372  vector<double> DDD = count_triplets_SphericalHarmonics(r12_min, r12_max, r13_min, r13_max, norders, catalogue);
373  coutCBL << "Done!" << endl;
374  coutCBL << endl;
375  coutCBL << "Counting triplet multipoles RRR" << endl;
376  vector<double> RRR = count_triplets_SphericalHarmonics(r12_min, r12_max, r13_min, r13_max, norders, random_catalogue);
377  coutCBL << "Done!" << endl;
378  coutCBL << endl;

381  coutCBL << "Done!" << endl;
382  coutCBL << endl;
383 
384  for (int l=0; l<norders; l++) {
385  RRR[l] *= (2.*l+1)/2;
386  NNN[l] *= (2.*l+1)/2;
387  }
388 
389 
390  // ------------------------------------------------------
391  // ---------------- compute the matrix A ----------------
392  // ------------------------------------------------------
393 
394  vector<double> fl = RRR;
395  for (int i=0; i<norders; i++)
396  fl[i] = fl[i]/RRR[0];
397

399 
400  for (int k=0; k<norders; k++)
401  for (int l=0; l<norders; l++)

404 
405  for (int k=0; k<norders; k++)

407 
408  invert_matrix(A, A_inverse);
409 
410 
411  // ----------------------------------------------------
412  // ---------------- compute the zeta_l ----------------
413  // ----------------------------------------------------
414 
415  vector<double> zeta_l(norders, 0);
416 
417  for (int l=0; l<norders; l++)
418  for (int k=0; k<norders; k++)
419  zeta_l[l] += NNN[k]*A_inverse[l][k]/RRR[0];
420 
421 
422  // --------------------------------------------------
423  // ---------------- compute the zeta ----------------
424  // --------------------------------------------------
425 
426  string mkdir = "mkdir -p "+output_dir;
427  if(system(mkdir.c_str())) {}
428 
429  const string file = output_dir+output_file;
430  ofstream fout(file.c_str());
431  vector<double> zeta(nbins, 0);
432 
433  for (int i=0; i<nbins; i++) {
434 
435  double angle = (i+0.5)/nbins*par::pi;
436  fout << setprecision(10) << angle/par::pi;
437  for (int l=0; l<norders; l++){

439  fout << setprecision(10) << " " << zeta[i];

441 
442  fout << endl;
443  }

762 
763  // start the multithreading parallelization
764 #pragma omp parallel num_threads(omp_get_max_threads())
765  {

767  shared_ptr<triplets::Triplet> tt_thread = move(triplets::Triplet::Create(tripletType, r12, r12_binSize, r13, r13_binSize, nbins));

791  double zz = jzz-izz;

810  }
811  }
812
