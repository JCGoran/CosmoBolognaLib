54 shared_ptr<cbl::catalogue::Catalogue> cbl::measure::threept::ThreePointCorrelation_comoving_multipoles::m_join_catalogues (const cbl::catalogue::Catalogue& data, const cbl::catalogue::Catalogue& random, const size_t startPos, const size_t endPos) const
55 {
56  // ---------------- check inputs ---------------------------------
57 
58  if (startPos > random.nObjects() || endPos > random.nObjects())
59  ErrorCBL("input indeces are larger than random catalogue size!", "m_join_catalogues", "ThreePointCorrelation_comoving_multipoles.cpp");
60 
61  // ---------------------------------------------------------------
62  // ---------------- construct the mixed catalogue ----------------
63  // ---------------------------------------------------------------
64 
65  double fact = 0;
66  for (size_t i=startPos; i<endPos; i++)
67  fact += random.weight(i);
68 
69  fact = -double(data.weightedN())/fact;
70 
71  catalogue::Catalogue joined_catalogue;
72 
73  for (size_t i=0; i<data.nObjects(); i++) {
74  double xx = data.xx(i);
75  double yy = data.yy(i);
76  double zz = data.zz(i);
77  double ww = data.weight(i);
78 
79  auto obj = make_shared<catalogue::Object>(catalogue::Object());
80  obj->set_xx(xx);
81  obj->set_yy(yy);
82  obj->set_zz(zz);
83  obj->set_weight(ww);
84  joined_catalogue.add_object(obj);
85  }
86 
87  for (size_t i=startPos; i<endPos; i++) {
88  double xx = random.xx(i);
89  double yy = random.yy(i);
90  double zz = random.zz(i);
91  double ww = random.weight(i)*fact;
92 
93  auto obj = make_shared<catalogue::Object>(catalogue::Object());
94  obj->set_xx(xx);
95  obj->set_yy(yy);
96  obj->set_zz(zz);
97  obj->set_weight(ww);
98  joined_catalogue.add_object(obj);
99  }
100 
101  return make_shared<Catalogue>(joined_catalogue);
102 }

108 vector<double> cbl::measure::threept::ThreePointCorrelation_comoving_multipoles::m_SzapudiSzalay_multipoles (const std::vector<double> nnn, const std::vector<double> rrr, const double normalization) const
109 {
110  vector<double> _NNN = nnn;
111  vector<double> _RRR = rrr;
112 
113  for (size_t i=0; i<m_nOrders; i++) {
114  _NNN[i] *= 0.5*(2.*i+1.);
115  _RRR[i] *= 0.5*(2.*i+1.);
116  }
117 
118  vector<double> fl = _RRR;
119  for (size_t i=1; i<m_nOrders; i++) {
120  fl[i] = fl[i]/_RRR[0];
121  }
122 
123  vector<vector<double>> A(m_nOrders, vector<double>(m_nOrders, 0)), A_inverse;
124 
125  for (size_t k=0; k<m_nOrders; k++)
126  for (size_t l=0; l<m_nOrders; l++)
127  for (size_t lp=1; lp<m_nOrders; lp++)
128  A[k][l] += (2*k+1)*pow(gsl_sf_coupling_3j(2*l, 2*lp, 2*k, 0, 0, 0), 2)*fl[lp];
129 
130  for (size_t k=0; k<m_nOrders; k++)
131  A[k][k] += 1.;
132 
133  invert_matrix(A, A_inverse, 1.e-4);
134 
135  vector<double> zeta_l(m_nOrders, 0.);
136  for (size_t l=0; l<m_nOrders; l++)
137  for (size_t k=0; k<m_nOrders; k++)
138  zeta_l[l] += _NNN[k]*A_inverse[l][k]/_RRR[0]*normalization;
139 
140  return zeta_l;
141 }

147 void cbl::measure::threept::ThreePointCorrelation_comoving_multipoles::set_catalogues (cbl::catalogue::Catalogue catalogue, cbl::catalogue::Catalogue random_catalogue, const double split, const int seed)
148 {
149  m_data = std::make_shared<catalogue::Catalogue>(catalogue::Catalogue(std::move(catalogue)));
150  m_random = std::make_shared<catalogue::Catalogue>(catalogue::Catalogue(std::move(random_catalogue)));
151  m_splitFactor = split;
152 
154  if (m_splitFactor>0) {
155  m_random->shuffle(seed);
156  double nSplit = static_cast<double>(m_random->nObjects())/(m_data->nObjects()*m_splitFactor);
157 
158  double integerPart, fracPart;
159  std::modf(nSplit, &integerPart);
160  fracPart = nSplit-integerPart;
161  m_nSplit = static_cast<size_t> (integerPart);
162 
163  if (integerPart<1) {
164  ErrorCBL("random catalogue too small!", "set_catalogues", "ThreePointCorrelation_comoving_multipoles.cpp");
165  } else if (fracPart > 0) {
166  WarningMsgCBL("a fraction of random objects will not be used; the effective size of the random sample will be "+conv(m_nSplit*m_data->nObjects()*m_splitFactor, par::fINT)+"!="+conv(m_random->nObjects(), par::fINT), "set_catalogues", "ThreePointCorrelation_comoving_multipoles.cpp");
167  }
168 
169  coutCBL << "Splitting activated! Three Point correlation function will be computed "<< m_nSplit << " times. Each time the random sample will have " << m_data->nObjects()*m_splitFactor << " objects." << endl;
170 
171  }
172  else
173  m_nSplit = 1;
174 
175 }
