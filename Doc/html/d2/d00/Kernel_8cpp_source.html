152 double cbl::round_to_digits (const double num, const int ndigits)
153 {
154  int exp_base10 = round(log10(num));
155  double man_base10 = num*pow(10., -exp_base10);
156  double factor = pow(10., -ndigits+1);
157  double truncated_man_base10 = man_base10-fmod(man_base10, factor);
158  double rounded_remainder = fmod(man_base10, factor)/factor;
159 
160  rounded_remainder = rounded_remainder > 0.5 ? 1.0*factor : 0.0;
161 
162  return (truncated_man_base10+rounded_remainder)*pow(10.0, exp_base10);
163 }

273 int cbl::check_memory (const double frac, const bool exit, const std::string func, const int type)
274 {
275 #ifdef LINUX
276  struct sysinfo memInfo;
277  sysinfo (&memInfo);
278 
279  long long freePhysMem = memInfo.freeram;
280  freePhysMem *= memInfo.mem_unit;
281  int used = used_memory(type);
282 
283  if (used > freePhysMem*0.001*frac) { // 0.001 is to convert kbytes in bytes
284  string Err = "Attention: possible memory problem";
285  Err += (func.empty()) ? "!\n" : " in "+func+" !\n";
286  Err += "freePhysMem = "+conv((double)freePhysMem*1.e-9, par::fDP3)+" GB\n";
287  Err += "memory used by the process: = "+conv((double)used*1.e-6, par::fDP3)+" GB\n";
288  if (exit) ErrorCBL(Err);
289  else { WarningMsg(Err); return 0; }
290  }
291  return 1;
292 
293 #else
294  (void)frac; (void)exit; (void)func; (void)type;
295  //WarningMsg("Attention: check_memory of Func.cpp works only on Linux systems");
296  return 1;
297 
298 #endif
