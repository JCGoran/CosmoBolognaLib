46 vector<double> cbl::fit_covariance_matrix_2PCF_monopole (const vector<double> mean, const vector<vector<double>> mock_xi0, const bool doJK, const cbl::cosmology::Cosmology cosmology, const double nObjects, const double Volume, const double bias, const double redshift, const double rMin, const double rMax, const int nbins, const cbl::BinType bin_type, const string method_Pk, const double sigma_NL, const bool NL)
47 {
48  coutCBL << "Fitting the Covariance Matrix" << endl;
49  const int nmock = mock_xi0.size();
50 
51  double term = (doJK) ? nmock-1 : 1.;
52  vector<double> mock_mean(nbins, 0);
53 
54  if (mean.size()==0) {
55  vector<vector<double>> vv = transpose(mock_xi0);
56  for (int i=0; i<nbins; i++)
57  mock_mean[i] = Average(vv[i]);
58  }
59  else
60  for (int i=0; i<nbins; i++)
61  mock_mean[i] = mean[i];
62 
63  vector<vector<double>> diff(nmock, vector<double>(nbins, 0));
64  for (int i=0; i<nmock; i++)
65  for (int j=0; j<nbins; j++)
66  diff[i][j] = mock_xi0[i][j]-mock_mean[j];
67 
68  cbl::cosmology::Cosmology cosmo = cosmology;
69  vector<double> kk = cbl::logarithmic_bin_vector(200, 1.e-4, 1.e2);
70  vector<double> Pk0, Pk2, Pk4;
71 
72  cosmo.Pk_Kaiser_multipoles (Pk0, Pk2, Pk4, kk, method_Pk, NL, redshift, bias, sigma_NL);
73 
74  vector<double> nObj = linear_bin_vector(20, 0.2*nObjects, nObjects);
75  vector<double> Vol = linear_bin_vector(20, 0.2*Volume, Volume);
76 
77  auto func = [&] (vector<double> &parameters)
78  {
79  const double _nObjects = parameters[0];
80  const double _Volume = parameters[1];
81  if (_nObjects<0.2*nObjects || _nObjects>nObjects ) return 1.e30;
82  if (_Volume<0.2*Volume || _Volume>Volume ) return 1.e30;
83 
84  vector<double> rr;
85  vector<vector<double>> covariance, icovariance;
86  cbl::Covariance_XiMultipoles (rr, covariance, nbins, rMin, rMax, _nObjects, _Volume, kk, {Pk0, Pk2, Pk4}, {0}, bin_type);
87  for (int i=0; i<nbins; i++)
88  for (int j=0; j<nbins; j++)
89  covariance[i][j] /=term;
90 
91  invert_matrix(covariance, icovariance);
92  double chi2 = 0.;
93  for (int i=0; i<nmock; i++)
94  chi2+=v_M_vt(diff[i], icovariance);
95  double LL = nmock*nbins*nbins*log(2*par::pi)+nmock*log(determinant_matrix(covariance))+chi2;
96  return LL;
97  };
98 
99  double min_nObj = 0., min_Vol = 0.;
100  double max_nObj = 50*nObjects, max_Vol = 50*Volume;
101  double min_LL = 1.e30;
102  for (size_t i=0; i<nObj.size(); i++)
103  for (size_t j=0; j<Vol.size(); j++) {
104  vector<double> pars = {nObj[i], Vol[j]};
105  double LL = func(pars);
106  if (LL<min_LL) {
107  min_LL = LL;
108  min_nObj = nObj[i];
109  min_Vol = Vol[j];
110  }
111  }
112 
113  vector<double> start = {nObjects, Volume};
114  vector<vector<double>> limits = { {min_nObj, max_nObj}, {min_Vol, max_Vol}};
115  return cbl::gsl::GSL_minimize_nD(func, start, limits);
116 }
