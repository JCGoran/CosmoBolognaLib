74 long cbl::chainmesh::ChainMesh::pos_to_index (const vector<double> center) const
75 {
76  vector<long> indx(m_nDim);
77 
78  for (long j=0; j<m_nDim; j++)
79  indx[j] = min(long((center[j]-m_Lim[j][0])/m_cell_size), m_nCell[j]-1);
80 
81  long indx_tot = indx[m_nDim-1];
82 
83  for (long j=m_nDim-2; j>-1; j--) {
84  long mult = 1;
85  for (long k=j+1; k<m_nDim; k++) mult *= m_nCell[k];
86  indx_tot += mult*indx[j];
87  }
88 
89  return indx_tot;

113 void cbl::chainmesh::ChainMesh::index_to_inds (const long index, const vector<long> nn, vector<long> &indx) const
114 {
115  indx.resize(m_nDim, 0);
116  long mult = 1;
117  long sum = 0;
118  for (int i=1; i<m_nDim; i++) mult *= nn[i];
119  indx[0] = floor((index-sum)/mult);
120  sum = indx[0]*mult;
121 
122  for (int i=1; i<m_nDim; i++) {
123  mult = 1;
124  for (int k=i+1; k<m_nDim; k++) mult *= nn[k];
125  indx[i] = floor((index-sum)/mult);
126  sum += indx[i]*mult;
127  }

134 void cbl::chainmesh::ChainMesh::create_chain_mesh (const vector<vector<double> > data, const double rMAX, const long nMIN, const long nMAX)
135 {
136  // setting stuff, generalized for n(=nDim) dimensions
137 
138  long nObj = data[0].size();
139  m_List.erase(m_List.begin(), m_List.end()); m_List.resize(nObj, -1);
140  m_NonEmpty_Cells.erase(m_NonEmpty_Cells.begin(), m_NonEmpty_Cells.end());
141 
142  m_nCell_tot = pow(nMAX,3)+1;
143  m_nCell_NonEmpty = 0;
144 
145  double fact = 1.;
146 
147  check_memory(3.0, true, "cbl::chainmesh::ChainMesh::create_chain_mesh of ChainMesh.cpp");
148 
149  while (m_nCell_tot>pow(nMAX, 3) || m_nCell_tot<nMIN
150  || !check_memory(3.0, false, "cbl::chainmesh::ChainMesh::create_chain_mesh of ChainMesh.cpp")) {
151 
152  m_nCell_tot = 1;
153  m_cell_size *= fact;
154  for (int i=0; i<m_nDim; i++) {
155  m_Lim[i][0] = Min(data[i])-1.05*rMAX;
156  m_Lim[i][1] = Max(data[i])+1.05*rMAX;
157  m_Delta[i] = m_Lim[i][1]-m_Lim[i][0];
158  m_nCell[i] = nint(m_Delta[i]/m_cell_size);
159  m_nCell_tot *= m_nCell[i];
160  m_cell_to_index[i] = m_Delta[i]/m_nCell[i];
161  }
162  fact *= 1.1;
163 
164  m_Label.erase(m_Label.begin(), m_Label.end());
165 
166  }
167 
168  m_Label.resize(m_nCell_tot, -1);
169 
170  for (long i=0; i<nObj; i++) {
171  vector<double> center(m_nDim);
172  for (int j=0; j<m_nDim; j++)
173  center[j] = data[j][i];
174  long indx_tot = pos_to_index(center);
175  m_NonEmpty_Cells.push_back(indx_tot);
176  m_List[i] = m_Label[indx_tot];
177  m_Label[indx_tot] = i;
178  }
179 
180  get_searching_region(rMAX);
181  m_NonEmpty_Cells = different_elements (m_NonEmpty_Cells);
182  m_nCell_NonEmpty = (long)m_NonEmpty_Cells.size();
183 
184 }

220 void cbl::chainmesh::ChainMesh::get_searching_region (const double r_max, const double r_min)
221 {
222  int n_max = nint(r_max/m_cell_size);
223  n_max = (n_max*m_cell_size<r_max) ? n_max+1 : n_max;
224 
225  int n_min = nint(r_min/m_cell_size)-2; // exclusive
226  vector<long> n_incl(m_nDim);
227 
228  for (int i=0; i<m_nDim; i++)
229  n_incl[i] = 2*n_max+1;
230 
231  long sz_region = pow(2*n_max+1, m_nDim);
232 
233  m_search_region.erase(m_search_region.begin(),m_search_region.end());
234  m_search_region.resize(sz_region,0);
235 
236  for (long i=0; i<sz_region; i++) {
237  vector<long> indx(m_nDim);
238  index_to_inds(i,n_incl,indx);
239  for (int i=0; i<m_nDim; i++) indx[i]-=n_max;
240  m_search_region[i] = inds_to_index(indx);
241  }
242 
243  if (r_min >0 && n_min>0) {
244  vector<long> n_excl(m_nDim);
245  for (int i=0; i<m_nDim; i++)
246  n_excl[i] = 2*n_min+1;
247 
248  long sz_excl_region = pow(2*n_min+1,m_nDim);
249 
250  for (long i=0; i<sz_excl_region; i++) {
251  vector<long> indx(m_nDim);
252  index_to_inds(i, n_excl, indx);
253  for (int i=0; i<m_nDim; i++) indx[i] -= n_min;
254  long veto = inds_to_index(indx);
255  m_search_region.erase(remove(m_search_region.begin(), m_search_region.end(), veto), m_search_region.end());
256  }
257  }
258 }

265 {

267  // r2 == -1 ---> search in a nDim sphere from center to r1
