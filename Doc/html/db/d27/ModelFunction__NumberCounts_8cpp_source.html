64 void cbl::modelling::numbercounts::sigmaM_dlnsigmaM (double &sigmaM, double &dlnsigmaM, const double mass, const cbl::glob::FuncGrid interp_Pk, const double kmax, const double rho)
65 {
66  double norm = 1./(2.*pow(par::pi, 2));
67  double dRdM_fact = pow(3./(4.*par::pi*rho), 1./3.);
68 
69  double RR = Radius(mass, rho);
70  double dRdM = dRdM_fact*pow(mass, -2./3.)/3.;
71 
72  auto integrand_sigmaR = [&] (const double kk)
73  {
74  return kk*kk*interp_Pk(kk)*Filter_sigmaR(kk, RR);
75  };
76 
77  sigmaM = norm*cbl::wrapper::gsl::GSL_integrate_cquad(integrand_sigmaR, 1.e-4, kmax, 1.e-5);
78 
79  auto integrand_dsigmaR = [&] (const double kk)
80  {
81  return kk*kk*interp_Pk(kk)*Filter_dsigmaR(kk, RR);
82  };
83 
84  dlnsigmaM = norm*cbl::wrapper::gsl::GSL_integrate_cquad(integrand_dsigmaR, 1.e-4, kmax, 1.e-5)*dRdM*(mass/(2*sigmaM));
85  sigmaM = sqrt(sigmaM);
86 }

293 double cbl::modelling::numbercounts::counts_proxy (double (*fz)(std::vector<double>, std::shared_ptr<void>), double (*z_error)(std::vector<double>), double (*proxy_error)(std::vector<double>), double (*transfer_func)(const double, const double, const double, const std::string, const double, const std::string, std::shared_ptr<void>), const double redshift_min, const double redshift_max, const double proxy_min, const double proxy_max, cbl::cosmology::Cosmology cosmology, cbl::catalogue::Cluster cluster, const double Area, const std::string model_MF, const std::string model_bias, const bool store_output, const double Delta, const bool isDelta_vir, const cbl::glob::FuncGrid interp_sigmaM, const cbl::glob::FuncGrid interp_DlnsigmaM, const double proxy_pivot, const double z_pivot, const double mass_pivot, const double log_base, const double weight)

304  {

306  double log_f_z = log( fz({the_redsh, z_pivot}, cosmology_ptr) )/log(log_base);

421  double mean = cluster.alpha_scaling_rel() + cluster.beta_scaling_rel()*log_M + cluster.gamma_scaling_rel()*log_f_z + log(proxy_pivot)/log(log_base);
422  double sigma = cluster.scatter0_scaling_rel() + cluster.scatterM_scaling_rel()*pow(log_M, cluster.scatterM_exponent_scaling_rel()) + cluster.scatterz_scaling_rel()*pow(log_f_z, cluster.scatterz_exponent_scaling_rel());
423  double P_lambda__M_z = (cbl::gaussian(log(x[2])/log(log_base), pp, {mean,sigma})) / (x[2]*log(log_base));
424 
425  // Compute the integrals of P(z|z) and P(lambda|lambda)
426  double mean_Pz = x[1] + cluster.zbias() * (1+x[1]);
427  double int_P_z = 0.5 * ( erf( (redshift_max - mean_Pz) / (sqrt(2)*z_error({cluster.zerror(), redshift_max})) ) - erf( (redshift_min - mean_Pz) / (sqrt(2)*z_error({cluster.zerror(), redshift_min})) ) );
428  double int_P_lambda = 0.5 * ( erf( (proxy_max - x[2]) / (sqrt(2)*proxy_error({cluster.proxyerror(), proxy_max})) ) - erf( (proxy_min - x[2]) / (sqrt(2)*proxy_error({cluster.proxyerror(), proxy_min})) ) );
429 
430  return transfer_func(Mass, interp_sigmaM(Mass*fact), x[1], model_bias, Delta_, "EisensteinHu", cosmology_ptr) * cosmology.mass_function(Mass, interp_sigmaM(Mass*fact), interp_DlnsigmaM(Mass*fact), x[1], model_MF, store_output, cbl::par::defaultString, Delta_)*Area*cosmology.dV_dZdOmega(x[1], true) * normM * P_lambda__M_z * int_P_z * int_P_lambda;
431  };
432 
433  // -------------------------------------------------------------
434 
435  // Find the minimum and maximum masses, given the parameters of the scaling relation
436  double scatter = cluster.scatter0_scaling_rel();
437 
438  double log_lambda_min = std::max(log(proxy_min/proxy_pivot)/log(log_base) - 3.5*scatter, log(0.0001/proxy_pivot)/log(log_base));
439  double log_lambda_max = log(proxy_max/proxy_pivot)/log(log_base) + 3.5*scatter;
440  double log_f_z_min = log( fz({redshift_min, z_pivot}, cosmology_ptr) )/log(log_base);
441  double log_f_z_max = log( fz({redshift_max, z_pivot}, cosmology_ptr) )/log(log_base);
442 
443  double M1 = (- cluster.alpha_scaling_rel() + log_lambda_min - cluster.gamma_scaling_rel()*log_f_z_min) / cluster.beta_scaling_rel();
444  double M2 = (- cluster.alpha_scaling_rel() + log_lambda_max - cluster.gamma_scaling_rel()*log_f_z_min) / cluster.beta_scaling_rel();
445  double M3 = (- cluster.alpha_scaling_rel() + log_lambda_min - cluster.gamma_scaling_rel()*log_f_z_max) / cluster.beta_scaling_rel();
446  double M4 = (- cluster.alpha_scaling_rel() + log_lambda_max - cluster.gamma_scaling_rel()*log_f_z_max) / cluster.beta_scaling_rel();
447 
448  double min1 = std::min(M1, M2);
449  double min2 = std::min(min1, M3);
450  double minM_ = std::min(min2, M4);
451  double max1 = std::max(M1, M2);
452  double max2 = std::max(max1, M3);
453  double maxM_ = std::max(max2, M4);
454 
455  double minM = std::max(minM_, log(1.e12/mass_pivot)/log(log_base));
456  double maxM = std::min(maxM_, log(1.e16/mass_pivot)/log(log_base));
457 
458  // Define the integral limits
459  int integral_dimension=3;
460  std::vector<std::vector<double>> integration_limits(integral_dimension);
461  integration_limits[0] = {minM, maxM};
462  integration_limits[1] = {std::max(redshift_min - 3.5*z_error({cluster.zerror(), redshift_min}), 0.), redshift_max + 3.5*z_error({cluster.zerror(), redshift_max})};
463  integration_limits[2] = {std::max(proxy_min - 3.5*proxy_error({cluster.proxyerror(), proxy_min}), 1.), proxy_max + 3.5*proxy_error({cluster.proxyerror(), proxy_max})};
464 
465  // Compute the integral
466  cbl::wrapper::cuba::CUBAwrapper CW (integrand, integral_dimension);
467  double nc;
468 
469  if (integration_limits[0][0] < integration_limits[0][1])
470  nc = CW.IntegrateVegas(integration_limits,false);
471  else
472  nc = 0;
473 
474  return nc * mass_pivot * log(log_base) * weight;
475 }
