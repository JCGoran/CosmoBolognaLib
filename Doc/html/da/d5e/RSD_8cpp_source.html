49 double cbl::cosmology::Cosmology::linear_growth_rate (const double redshift, const double kk, const double prec) const
50 {
51  // Kiakotou, Elgarøy & Lahav 2008
52  double fnu = m_Omega_neutrinos/m_Omega_matter;
53  double a_in = 1.e-8;
54  double ff = (m_Omega_radiation==0.) ? 1. : (a_in/m_Omega_radiation*m_Omega_matter)/(a_in/m_Omega_radiation*m_Omega_matter+2./3.); // Dodelson eq. (7.59)
55  double w = m_w0+m_wa*redshift/(redshift+1.); // CPL parameterisation
56 
57  if (fnu<=0 or kk<0) {
58 
59  auto func = [&] (const double y, double &dyda, const double ln_aa)
60  {
61  const double zz = 1./exp(ln_aa)-1.;
62  const double OmM = OmegaM(zz);
63  dyda = -y*y-y*(1.-0.5*(OmM+OmegaR(zz)+(1.+3.*m_w0+3.*m_wa*zz/(1.+zz))*OmegaDE(zz)))+1.5*OmM;
64  };
65 
66  boost::numeric::odeint::integrate_adaptive( make_controlled(1e-8, 1e-8, stepper_type() ),
67  func, ff, log(a_in), log(1./(1.+redshift)), prec);
68  return ff;
69  }
70 
71  else {
72 
73  if (redshift>10 or w>-0.5) WarningMsgCBL("the approximation to compute the linear growth rate is not very accurate at z>10 or at w>-0.5 (see Kiakotou, Elgaroy & Lahav 2008)!", "linear_growth_rate", "RSD.cpp");
74  if (kk<0) ErrorCBL("kk<0!", "linear_growth_rate", "RSD.cpp");
75 
76  // Wang & Steinhardt 1998
77  double alpha0 = 3./(5.-w/(1.-w));
78  double alpha1 = 3./125.*(1.-w)*(1.-3.*w/2.)/pow(1.-6.*w/5.,3);
79  double alpha = alpha0+alpha1*(1-OmegaM(redshift));
80  double param = (alpha-4./7.)*m_Omega_k; // Gong et al. 2009
81 
82  double kkk[] = {0.001, 0.01, 0.05, 0.07, 0.1, 0.5};
83  double aa[] = {0., 0.132, 0.613, 0.733, 0.786, 0.813};
84  double bb[] = {0., 1.62, 5.59, 6., 5.09, 0.803};
85  double cc[] = {0., 7.13, 21.13, 21.45, 15.5, -0.844};
86 
87  vector<double> lgKK;
88  //lgKK.push_back(log10(0.005)); // to improve the interpolation
89  for (int i=0; i<6; i++) lgKK.push_back(log10(kkk[i]));
90 
91  //vector<double> KK;
92  //KK.push_back(0.005); // to improve the interpolation
93  //for (int i=0; i<6; i++) KK.push_back(kkk[i]);
94 
95  vector<double> MU;
96 
97  for (int i=0; i<6; i++) MU.push_back(1.-aa[i]*m_Omega_DE*fnu+bb[i]*fnu*fnu-cc[i]*fnu*fnu*fnu);
98 
99  double mu;
100 
101  if (kk<0.001) mu = interpolated(log10(kk), lgKK, MU, "Poly"); // same?
102  else if (0.001<=kk && kk<=0.5) mu = interpolated(log10(kk), lgKK, MU, "Poly");
103 
104  else mu = pow(1-fnu, alpha0);
105 
106  ff = mu*pow(OmegaM(redshift), alpha) + param; // Gong et al. 2009
107  }
108 
109  return ff;
110 }
