46 double cbl::cosmology::Cosmology::linear_growth_rate (const double redshift, const double kk) const
47 {
48  if (m_wa!=0) WarningMsg("Attention in cbl::cosmology::Cosmology::linear_growth_rate of RSD.cpp: the current implementation is not correct with m_wa="+conv(m_wa, par::fDP3));
49 
50  if (redshift>10) WarningMsg("Attention: the approximation used in cbl::cosmology::Cosmology::linear_growth_rate of RSD.cpp to compute the linear growth rate is not very accurate at z>10 (see Kiakotou, Elgaroy & Lahav 2008)!");
51 
52 
53  // Wang & Steinhardt 1998
54 
55  double alpha0 = 3./(5.-m_w0/(1.-m_w0));
56  double alpha1 = 3./125.*(1.-m_w0)*(1.-3.*m_w0/2.)/pow(1.-6.*m_w0/5.,3);
57  double alpha = alpha0+alpha1*(1-OmegaM(redshift));
58 
59 
60  // Kiakotou, Elgarøy & Lahav 2008
61 
62  double fnu = m_Omega_neutrinos/m_Omega_matter;
63  double ff = -1.;
64 
65  if (fnu>0) {
66  if (kk<0) ErrorCBL("Error in cbl::cosmology::Cosmology::linear_growth_rate of RSD.cpp: kk<0!");
67 
68  double kkk[] = {0.001, 0.01, 0.05, 0.07, 0.1, 0.5};
69  double aa[] = {0., 0.132, 0.613, 0.733, 0.786, 0.813};
70  double bb[] = {0., 1.62, 5.59, 6., 5.09, 0.803};
71  double cc[] = {0., 7.13, 21.13, 21.45, 15.5, -0.844};
72 
73  vector<double> lgKK;
74 //lgKK.push_back(log10(0.005)); // to improve the interpolation
75  for (int i=0; i<6; i++) lgKK.push_back(log10(kkk[i]));
76 
77  //vector<double> KK;
78  //KK.push_back(0.005); // to improve the interpolation
79  //for (int i=0; i<6; i++) KK.push_back(kkk[i]);
80 
81  vector<double> MU;
82  // MU.push_back(1.);
83  for (int i=0; i<6; i++) MU.push_back(1.-aa[i]*m_Omega_DE*fnu+bb[i]*fnu*fnu-cc[i]*fnu*fnu*fnu);
84 
85  double mu;
86 
87  if (kk<0.001) mu = interpolated(log10(kk), lgKK, MU, "Poly");
88  else if (0.001<=kk && kk<=0.5) mu = interpolated(log10(kk), lgKK, MU, "Poly");
89 
90  else mu = pow(1-fnu, alpha0);
91 
92  ff = mu*pow(OmegaM(redshift), alpha);
93  }
94 
95  else
96  ff = pow(OmegaM(redshift), alpha);
97 
98  ff += (alpha-4./7.)*m_Omega_k; // Gong et al. 2009
99 
100  return ff;
101 }
