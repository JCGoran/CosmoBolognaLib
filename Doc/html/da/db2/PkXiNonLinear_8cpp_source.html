92 double cbl::cosmology::Cosmology::Pk_1loop (const double kk, const shared_ptr<cbl::glob::FuncGrid> Pk, const int corrtype, const double qmin, const double qmax, const double prec)
93 {
94  function<double(double, double, double)> func1, func2;
95 
96  if(corrtype==0) {
97  func1 = [&] (const double k, const double q, const double kq) {return F2(k, q, kq);};
98  func2 = [&] (const double k, const double q, const double kq) {return F2(k, q, kq);};
99  }
100  else if(corrtype==1) {
101  func1 = [&] (const double k, const double q, const double kq) {return F2(k, q, kq);};
102  func2 = [&] (const double k, const double q, const double kq) {return G2(k, q, kq);};
103  }
104  else if(corrtype==2) {
105  func1 = [&] (const double k, const double q, const double kq) {return G2(k, q, kq);};
106  func2 = [&] (const double k, const double q, const double kq) {return G2(k, q, kq);};
107  }
108  else
109  ErrorCBL("Error in Pk_1loop() of PkXiNonLinear! You must choose a correct value for corrtype!");
110 
111  auto integrand = [&] (const double qq)
112  {
113  auto integrand_intermediate = [&] (const double xx)
114  {
115  double kq = sqrt(kk*kk+qq*qq-2.*kk*qq*xx);
116  double akq = (kk*xx-qq)/kq;
117  return func1(kq, qq, akq)*func2(kq, qq, akq)*Pk->operator()(kq);
118  };
119  double res = gsl::GSL_integrate_qag(integrand_intermediate, -1, 1, prec);
120  return Pk->operator()(qq)*qq*qq*res;
121  };
122 
123  return 4.*par::pi*gsl::GSL_integrate_qag(integrand, qmin, qmax, prec);
124 }
